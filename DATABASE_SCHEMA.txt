"""
Crime Report Portal - Database Schema
SQLite Database Models
"""

# This file demonstrates the database schema structure

# Users Table
# ============
# id: INTEGER PRIMARY KEY
# name: VARCHAR(100) NOT NULL
# email: VARCHAR(120) UNIQUE NOT NULL
# password_hash: VARCHAR(255) NOT NULL
# phone: VARCHAR(20) NOT NULL
# role: VARCHAR(20) DEFAULT 'user'  # 'user' or 'admin'
# created_at: DATETIME
# last_login: DATETIME
# is_verified: BOOLEAN DEFAULT False

# Complaints Table
# ================
# id: INTEGER PRIMARY KEY
# user_id: INTEGER FK(users.id)
# category: VARCHAR(50) NOT NULL
# title: VARCHAR(200) NOT NULL
# description: TEXT NOT NULL
# location: VARCHAR(255) NOT NULL
# latitude: FLOAT
# longitude: FLOAT
# incident_date: DATETIME NOT NULL
# status: VARCHAR(50) DEFAULT 'Submitted'
# priority: VARCHAR(20) DEFAULT 'Medium'
# created_at: DATETIME
# updated_at: DATETIME

# Evidence_Files Table
# ====================
# id: INTEGER PRIMARY KEY
# complaint_id: INTEGER FK(complaints.id)
# filename: VARCHAR(255) NOT NULL
# file_path: VARCHAR(255) NOT NULL
# file_type: VARCHAR(20) NOT NULL
# file_size: INTEGER
# uploaded_at: DATETIME

# Tracking_Updates Table
# ======================
# id: INTEGER PRIMARY KEY
# complaint_id: INTEGER FK(complaints.id)
# status: VARCHAR(50) NOT NULL
# message: TEXT NOT NULL
# created_at: DATETIME

# Management_Notes Table
# ======================
# id: INTEGER PRIMARY KEY
# complaint_id: INTEGER FK(complaints.id)
# admin_id: INTEGER FK(users.id)
# action: VARCHAR(50)
# note: TEXT NOT NULL
# created_at: DATETIME

# Notifications Table
# ===================
# id: INTEGER PRIMARY KEY
# user_id: INTEGER FK(users.id)
# complaint_id: INTEGER FK(complaints.id)
# title: VARCHAR(200) NOT NULL
# message: TEXT NOT NULL
# notification_type: VARCHAR(50)
# is_read: BOOLEAN DEFAULT False
# sent_via: VARCHAR(50) DEFAULT 'portal'
# created_at: DATETIME
